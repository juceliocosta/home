Exemplo de Eventos de Mouse

<script>

        // Seleciona o botão pelo seu ID

        var botao = document.getElementById('meuBotao');

        // Adiciona um ouvinte de evento para o evento 'mouseover'

        botao.addEventListener('mouseover', function() {

            botao.style.backgroundColor = 'lightblue';

            botao.textContent = 'Mouse está aqui!';

        });

        // Adiciona um ouvinte de evento para o evento 'mouseout'

        botao.addEventListener('mouseout', function() {

            botao.style.backgroundColor = '';

            botao.textContent = 'Passe o mouse ou clique';

        });

        // Adiciona um ouvinte de evento para o evento 'click'

        botao.addEventListener('click', function() {

            alert('Botão clicado!');

        });

    </script>
