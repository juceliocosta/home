Exemplo de Métodos JavaScript

<script>

        // Função para exibir um alert

        function exibirAlert() {

            window.alert('Esta é uma mensagem de alerta!');

        }

        // Função para exibir uma confirmação

        function exibirConfirm() {

            var confirmacao = window.confirm('Você deseja continuar?');

            if (confirmacao) {

                alert('Você pressionou OK');

            } else {

                alert('Você pressionou Cancelar');

            }

        }

        // Função para exibir um prompt

        function exibirPrompt() {

            var texto = window.prompt('Por favor, insira seu nome:');

            if (texto !== null) {

                alert('Olá, ' + texto + '!');

            } else {

                alert('Nenhum nome foi inserido.');

            }

        }

        // Função para escrever no documento

        function escreverDocumento() {

            document.write('<p>Este texto foi adicionado ao documento usando document.write</p>');

        }

        // Função para registrar um log no console

        function registrarLog() {

            console.log('Esta é uma mensagem de log.');

        }

        // Função para registrar um aviso (warn) no console

        function registrarWarn() {

            console.warn('Esta é uma mensagem de aviso (warn).');

        }

        // Função para registrar um erro no console

        function registrarError() {

            console.error('Esta é uma mensagem de erro.');

        }

    </script>

Métodos mais utilizados

window.alert abre uma caixa de diálogo para mensagens (botão ok)

window.confirm exibe uma caixa de diálogo para confirmação (Ok/Cancelar)

window.prompt exibe uma caixa de diálogo para entrada de texto

document.write adiciona conteúdo no documento HTML

console.log registra conteúdo de log no console do navegador

console.warn registra mensagem de warning no console do navegador

console.error registra mensagem de erro no console do navegador
